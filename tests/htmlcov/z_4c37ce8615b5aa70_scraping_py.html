47def get_categories(): 

48 """ 

49 Cria um dicionário com as categorias

52 res = requests.get(URL) 

53 soup = BeautifulSoup(res.text, "lxml")

65def regex_key_book_info(key: str) -> str: 

66 """ 

67 Responsável por formatar o nome do campo da tabela de informações 

68 do produto para que ele vire uma chave adequada para o dicionário 

69 de informações sobre o livro. 

70 

71 Args: 

72 key (str): Nome da informação na tabela de informações do livro. 

73 

74 Returns: 

75 Nome formatado. 

76 """ 

77 # Substituir espaços e parênteses/pontos por underscore no 

78 # nome convertido em letras minúsculas. 

79 key = re.sub(r"[()\s\.]+", "_", key.lower()) 

80 return key.strip("_")

91def get_book_info(book_url: str) -> dict: 

92 """ 

93 Realiza o scraping da página específica do livro. 

94 

95 Args: 

96 book_url (str): URL do livro. 

97 

98 Retruns: 

99 Dicionário com as informações do livro. 

100 """ 

101 response = requests.get(book_url) 

102 soup = BeautifulSoup(response.text, 'lxml') 

103 infos = dict() 

104 

105 infos['title'] = soup.find('h1').text.strip() 

106 infos['rating'] = RATING_MAPPER[soup.select_one('.star-rating')['class'][1]] 

107 infos['category'] = soup.select('ul.breadcrumb li')[2].text.strip() 

108 

109 description_tag = soup.select_one('#product_description ~ p') 

110 infos['description'] = ( 

111 description_tag.text.strip() 

112 if description_tag else 'No description' 

113 ) 

114 

115 image_rel_url = soup.select_one('.item img')['src'] 

116 infos['image_url'] = urljoin(book_url, image_rel_url) 

117 

118 table = soup.select_one('table.table.table-striped') 

119 table_data = { 

120 regex_key_book_info(row.th.text.strip()): row.td.text.strip() 

121 for row in table.find_all('tr') 

122 } 

123 

124 infos = {**infos, **table_data} 

125 

126 infos['price_excl_tax'] = numbers_format(infos['price_excl_tax']) 

127 infos['price_incl_tax'] = numbers_format(infos['price_incl_tax']) 

128 infos['availability'] = int(re.search(r"\d+", infos['availability']).group()) 

129 infos['tax'] = numbers_format(infos['tax']) 

130 

131 return infos

134def get_total_pags(): 

135 """ 

136 Obtém o número total de páginas disponíveis no site base. 

137 

138 Returns: 

139 int: Quantidade total de páginas encontradas na paginação. 

140 """ 

141 url = f'{BASE_URL}page-1.html' 

142 res = requests.get(url) 

143 soup = BeautifulSoup(res.text, "lxml") 

144 paginacao = soup.select_one("li.current").get_text(strip=True) 

145 total_paginas = int(paginacao.split("of")[-1]) 

146 return total_paginas

149def scrape_books(pages: int = None) -> list: 

150 """ 

151 Varre o número de páginas passado como parametro pages para obter 

152 os links das páginas de cada livro e então chama a função que faz 

153 o web scraping da pagina do livro. 

154 

155 Args: 

156 pages (opcional, int): Número de páginas onde coletaremos os livros. 

157 

158 Returns: 

159 list: Lista de dicionários com as informações do livros presentes 

160 nas páginas que foram percorridas. 

161 """ 

162 pages = get_total_pags() if pages is None else pages 

163 books = [] 

164 for page in range(1, pages + 1): 

165 url = f'{BASE_URL}page-{page}.html' 

166 # Recupera a pagina com requests e cria um objeto scraping 

167 # com o BeautifulSoup e lxml por ser mais robusto. 

168 response = requests.get(url) 

169 soup = BeautifulSoup(response.text, 'lxml') 

170 book_links = soup.select('h3 a') 

171 for link in book_links: 

172 book_url = urljoin(BASE_URL, link['href']) 

173 book_data = get_book_info(book_url) 

174 books.append(book_data) 

175 

176 return books
